1. Wiggling Effect.

Code:

              
                .wigglingEffect {
                  width: 100px;
                  height: 100px;
                  background-color: #f00;
                  animation: wiggling 1s infinite;
                }

                @keyframes wiggling {
                  0% {
                    transform: rotate(0deg);
                  }
                  25% {
                    transform: rotate(5deg);
                  }
                  50% {
                    transform: rotate(0deg);
                  }
                  75% {
                    transform: rotate(-5deg);
                  }
                  100% {
                    transform: rotate(0deg);
                  }
                }
              
              
gsap code: 

              
                gsap.to(".wigglingEffect ", {
                    rotation: 10,
                    duration: 0.2,
                    repeat: -1,
                    yoyo: true,
                    ease: "power1.inOut",
                  });

3. Typing Text Effect.

Text: "Hello GSAP!"

Code:

              
                const text = "Hello, GSAP!";
                const el = document.querySelector(".typingTextEffect");
                let charIndex = 0;
                
                function type() {
                  if (charIndex < text.length) {
                    el.textContent += text[charIndex++];
                    setTimeout(type, 100);
                  }
                }
                
                setInterval(() => {
                  el.textContent = "";
                  charIndex = 0;
                  type();
                }, 2000);
